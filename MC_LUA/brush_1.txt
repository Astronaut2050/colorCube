-- 天花板基准方块
local CEILING_BLOCK = $blockState(天花板基准方块, stone)$

-- 两种方块及其权重（0~100）
local LAMP1 = $blockState(灯具1, glowstone)$
local LAMP1_WEIGHT = $int(灯具1比例, 50, 0, 100)$

local LAMP2 = $blockState(灯具2, shroomlight)$
local LAMP2_WEIGHT = $int(灯具2比例, 50, 0, 100)$

-- 灯带方向控制
local USE_X_DIRECTION = $boolean(灯带以X轴为方向, true)$

-- 灯带长度（1~10）
local STRIP_LENGTH = $int(灯带长度, 1, 1, 10)$

-- 灯间距控制：X方向步长和Z方向步长
local X_STEP = $int(X方向步长, 1, 1, 10)$
local Z_STEP = $int(Z方向步长, 1, 1, 10)$


-- 按权重选择一个方块：概率 = LAMPX_WEIGHT / (总和)
local function pick_lamp()
    local w1 = LAMP1_WEIGHT
    local w2 = LAMP2_WEIGHT

    local total = w1 + w2
    if total <= 0 then
        return LAMP2
    end

    -- 随机数 r
    local r = math.random() * total

    if r < w1 then
        return LAMP1
    else
        return LAMP2
    end
end

-- 当前位置的 BlockState（与 $blockState 参数对齐）
local current = getBlockState(x, y, z)

-- 当前是天花板基准方块时，在正下方生成带间距的灯矩阵
if current == CEILING_BLOCK then
    -- 检查当前位置是否在步长网格点上（即灯矩阵的起始点）
    if (x % X_STEP) == 0 and (z % Z_STEP) == 0 then
        local belowY = y - 1
        if belowY >= 0 then
            if USE_X_DIRECTION then
                -- 以X轴为方向：生成 X长度 × 1 的灯带
                for dx = 0, (STRIP_LENGTH - 1) do
                    local tx = x + dx
                    local belowBlock = getBlock(tx, belowY, z)
                    -- 只在当前位置是空气时才生成（避免重复生成）
                    if belowBlock == blocks.air then
                        setBlock(tx, belowY, z, pick_lamp())
                    end
                end
            else
                -- 以Z轴为方向：生成 1 × Z长度 的灯带
                for dz = 0, (STRIP_LENGTH - 1) do
                    local tz = z + dz
                    local belowBlock = getBlock(x, belowY, tz)
                    -- 只在当前位置是空气时才生成（避免重复生成）
                    if belowBlock == blocks.air then
                        setBlock(x, belowY, tz, pick_lamp())
                    end
                end
            end
        end
    end
end

-- 始终返回当前位置原方块
return current
